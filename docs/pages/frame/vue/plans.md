#### 图片加载失败后设置默认图片
```
<img  :src="$store.state.user.avatar" :onerror="avatarLoadError()" ref="avatar"/>
methods: {
    avatarLoadError(){
        this.$nextTick(function(){
            const img = this.$refs.avatar
            img.src = require('../assets/images/avatar-default.png')
            img.onerror=null
        })
    }
}

```
